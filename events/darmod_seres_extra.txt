namespace = seres_extra

#0
event = {
	id = seres_extra.1
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		years_passed >= 200  #>= mayor que / <= menor que		
		any_planet_within_border = { has_planet_flag = seres_extra_moon }
		NOT = { has_global_flag = seres_extra_site_spawned }		
		is_ai = no
		is_at_war = no
		is_subject = no	
	}

	immediate = {
		set_global_flag = seres_extra_site_spawned	
		country_event = { id = seres_extra.2 days = 1800 random = 3600 }
	}
}	

country_event = {
	id = seres_extra.2
	title = "seres_extra.2.name"
	desc = "seres_extra.2.desc"
	picture = GFX_evt_luna
	show_sound = event_power_down
	location = event_target:seres_extra_moon
	
	is_triggered_only = yes

	
	immediate = {
		set_global_flag = seres_extra_site_spawned
		random_planet_within_border = {
			limit = {
				has_planet_flag = seres_extra_moon
			}
			save_event_target_as = seres_extra_moon
		}
	}

	option = {
		name = seres_extra.2.a
		event_target:seres_extra_moon = {
			create_archaeological_site = DARH_seres_site_extra_1
		}
	}
}
