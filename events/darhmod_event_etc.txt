ship_event = {
	id = darhetc.3
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		#mid_game_years_passed > -60	
		has_global_flag = darhmod_precursor_set_seres
		owner = { is_ai = no }	
		from = {
			solar_system = {
				exists = star
				NOR = {
					has_star_flag = empire_cluster
					is_fe_cluster = yes
				}
				any_system_planet = {
					NOT = { exists = archaeological_site }
				}
			}
			has_anomaly = no
			OR = {
				is_colonizable = yes
				is_planet_class = pc_barren
				is_planet_class = pc_barren_cold
			}
			NOR = {
				exists = archaeological_site
				has_planet_flag = precursor_world
				has_global_flag = darhmod_precursor_seres_start
				has_planet_flag = pspmod_sless_colony
				has_planet_flag = pspmod_coprean_colony
				has_planet_flag = pspmod_astanine_colony
			}
		}
	}
	
	immediate = {
		from = {
			set_planet_flag = precursor_world
			planet_event = { id = darhetc.4 days = 3600 } #10 years after
		}
	}
}
