seres_anomaly_category = {
	desc = seres_anomaly_desc
	picture = GFX_evt_ship_seres
	level = 10
	max_once_global = yes
	
	spawn_chance = {
		base = 25
		modifier = {
			add = 100
			always = yes
			is_star = no
			is_asteroid = no			
			uninhabitable_regular_planet = yes
			NOT = {
				is_planet_class = pc_gas_giant			
			}
			has_anomaly = no
			from.owner = { is_ai = no }					
			OR = {
				NOT = { has_global_flag = darhmod_precursor_seres }
				has_global_flag = darhmod_precursor_seres_start
			}	
			solar_system = {
				exists = star
				NOR = {
					has_star_flag = empire_cluster
					is_fe_cluster = yes
				}
				any_system_planet = {
					NOT = { exists = archaeological_site }
				}
			}
		}
	}
	on_success = darhmod_seres.1
}
