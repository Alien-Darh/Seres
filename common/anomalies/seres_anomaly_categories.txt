		modifier = {

			is_star = no
			is_asteroid = no

				is_planet_class = pc_frozen
				is_planet_class = pc_toxic			
			}
			NOT = {
				is_planet_class = pc_gas_giant			
			}			
		}
		modifier = {
			factor = 0
			OR = {
				NOT = { has_global_flag = darhmod_precursor_seres }
				has_global_flag = darhmod_precursor_seres_start
			}
		}		
		modifier = {
			factor = 0
			solar_system = {
				OR = {
					NOT = { exists = star }
					has_star_flag = empire_cluster
					is_fe_cluster = yes
					any_system_planet = {
						OR = {
							exists = archaeological_site
							has_anomaly = yes
						}
					}
				}
			}
		}
	}
